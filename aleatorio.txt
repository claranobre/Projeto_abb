int contador(ArvoreBinariaBusca h){
		if(h == NULL)
			return 0;
		return contador(h->esquerda) + contador(h->direita) + 1;
	}

	/*
	* Altura de um nó 'h' em uma árvore binária
	*/

	int altura(ArvoreBinariaBusca h){
		int u, v;
		if(h == NULL)
			return -1;
		u = altura(h->esquerda);
		v = altura(h->direita);
		if(u > v)
			return u + 1;
		else
			return v + 1;
	}

		if(h == NULL) return;
		printf("%d \n", h->x);
		imprime(h->esquerda);
		imprime(h->direita);
